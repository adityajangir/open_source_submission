### PROGRAM TO CALCULATE SUM OF ASCII VALUES OF CHARACTERS 
```c++
#include <iostream>
using namespace std;
int main(){
    string name = "ABHINAV KUMAR CHOUDHARY";
    int l = name.length();
    int sum = 0;
    for(int i=0; i<l ; i++){
        char c = name[i];
        sum = sum + int(c);
    }
    cout << "The sum of ASCII values of characters of my name is : " << sum;

}

```
