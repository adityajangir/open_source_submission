#### **Code to print sum of ASCII values of a name : :**
```
#include <iostream>
using namespace std;

int main()
{  int i,sum;
   sum=0;
   
   string name = "M Samson";
   
   for(i=0;i<sizeof(name);i++)
   {
       sum+=int(name[i]);
   }
   
   cout<<"Sum of Ascii characters of "<<name<<" is "<<sum;
   return 0;
   
}
```
