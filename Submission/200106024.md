## Program which gives SUM of ASCII values of Full Name
***
```cpp
#include<bits/stdc++.h>

using namespace std;

int main()
{
	int sum=0;
	int i=0;
	string s="DAKSH KAUSHIK";
	for(i=0; i<s.size(); i++){
		if(s[i]==' ')
			continue;
		else
			sum+=s[i];
	}
	cout<<sum;
	return 0;
}
```
