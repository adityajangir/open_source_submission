>**Code To Print The Sum of ASCII Value of My Name:**

``` c++
#include <bits/stdc++.h>
using namespace std;
int main()
{
    int sum=0;
    string p;
    p="NITISH KUMAR";
    for(int i=0; i<p.length(); i++)
    {
        if(p[i]!=' ')
        sum+=p[i];
    }
    cout<<sum;
    return 0;
}
```
